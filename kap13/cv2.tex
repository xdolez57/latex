\def\obrazek#1#2{
\savebox{\obr}{\includegraphics{#1}}
\settowidth{\imagewidth}{\usebox{\obr}}
\settoheight{\imageheight}{\usebox{\obr}}
\commentwidth=\linewidth
\addtolength{\commentwidth}{-4mm}
\addtolength{\commentwidth}{-\imagewidth}
\addtolength{\imageheight}{-\baselineskip}
\begin{figure}[h]
\stepcounter{figure}
\noindent\raisebox{\imageheight}{\parbox[t]{\commentwidth}{Obr. \thefigure: #2}}\hspace*{\fill}\usebox{\obr}
\end{figure}}
